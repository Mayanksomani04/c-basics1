{

    length=str.length();
    for(int i=0;i<length;i++){
    if(i==0||i==(length-1)){
    str[i]=toupper(str[i]);
    }
    else 
    if(str[i]==' ')
    {
    str[i-1]=toupper(str[i-1]);
        str[i+1]=toupper(str[i+1]);

     }

}
