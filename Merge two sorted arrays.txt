Merge two sorted arrays

#include<iostream>
#include <cstring>

using namespace std;
int main(){
    int n,m;
    cout<<"Enter the valu of n and m ";
     cin>>n>>m;
 int arr1[n];
   cout<<"Enter the value of arr1 ";
 for(int i=0;i<n;i++){
    cin>>arr1[i];
 }
 int arr2[m];
   cout<<"Enter the valu of arr2  ";
 for(int i=0;i<m;i++){
     cin>>arr2[i];
 }
 int sum[n+m];
 
 int i=0,j=0,k=0;
  while (i < n && j < m) {
      if (arr1[i] < arr2[j]) {
         sum[k] = arr1[i];
         k++;
         i++;
      } else {
         sum[k] = arr2[j];
           k++;
         j++;
      }
   }
   while (j>=m && i < n) {
      sum[k] = arr1[i];
        k++;
         i++;
   }
   while (j < m && i>=n) {
      sum[k] = arr2[j];
        k++;
         j++;
   }
   cout<<" the merged array is ";
   for (int i = 0; i < n+m; ++i) {
     cout<<sum[i]<<" ";
   }
     return 0;
}
