Binary To Decimal
#include <iostream>

using namespace std;

int power(int number,int ex){
    int i=0,res=1;
    while(i<ex){
        res=res*number;
        i++;
    }
    return res;
}
int main(){
    int num,rem,count=0;
    cout<<"Enter a binary number ";
    cin>>num;
    int i=0,ans=0;
    while(num>0){
        rem=num%10;
        num=num/10;
        ans=ans+(rem*power(2,i));
        i++;
    }
    cout<<"The answer in decimal is ";
    cout<<ans;
    return 0;
}
