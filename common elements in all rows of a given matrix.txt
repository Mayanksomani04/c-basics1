
//Common elements in all rows of a given matrix
#include <iostream>
using namespace std;
int main(){
    int matrix[50][50];
    int r,c,i,j,k = 0,l = 0;
    cout<<"Enter number of rows and columns: ";
    cin>>r>>c;
    cout<<"Enter elements in matrix: ";
    for(i = 0; i < r; i++){
        for(j = 0; j < c; j++){
            cin>>matrix[i][j];
        }
    }
    cout<<"Common elements in all rows of a given matrix: ";
    for(k = 0; k < c; k++){
        int count = 1;
        for(i = 1; i < r; i++){
            for(j = 0; j < c; j++){
                if(matrix[i][j] == matrix[0][k]){
                    count++;
                    break;
                }
            }
        }
        if(count >= r){
            cout<<matrix[0][k];
        }
        else{
            l++;
        }
    }
    if(l > c - 1){
        cout<<"No common element is present";
    }
    return 0;
}
