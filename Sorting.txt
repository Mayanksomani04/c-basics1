Sorting

#include<iostream>
using namespace std;
int main(){
  int n;
  cout<<"Enter the size of array ";
  cin>>n;
  int arr[n];
  cout<<"Enter the  array  ";
  for(int i=0;i<n;i++){
      cin>>arr[i];
  }
  int temp=0,i,j;
   for( i=0;i<n;i++){
       for(int j=0;j<n;j++){
           if(arr[i]<arr[j]){
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
           }
       }
   }
   cout<<"The sorted array is "<<endl;
   for(int i=0;i<n;i++){
      cout<<arr[i];
  }
  return 0;
         }
