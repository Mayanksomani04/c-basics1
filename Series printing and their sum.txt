Write a program to print and find the sum of series of 1+x+x^2+x^3........
#include <iostream>
#include <cmath>

using namespace std;

int main(){
   
int n,x;
cout<<"Enter the value of n and x ";
cin>>n>>x;
int sum=1;
cout<<"1";
for(int i=1;i<n;i++){
    sum=sum+pow(x,i);
    cout<<"+"<<"x"<<"^"<<i;
    
}
cout<<endl;
cout<<sum;
    return 0;
}
