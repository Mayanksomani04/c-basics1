Printing by Pointer 
#include <iostream>
using namespace std;

int main()
{
  int n,i;
   cout<<"Enter the size of array ";
   cin>>n;
   int a[n];
   cout<<"Enter the elements of array "; 
   for( i=0;i<n;i++){
       cin>>a[i];
   }  
   int *p;
   p = &a[0];
   cout<<"The elements of array is ";
    while(p <= &a[n - 1]){
        cout<<*p<<" ";
        p++;
    }
    return 0;
}
