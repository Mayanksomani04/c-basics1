 3-D Multi-Sorting


#include<iostream>
using namespace std;
void sort(int arr[],int n){
    int temp=0,i=0,j;
   while(i<n){
       j=i+1;
       while(j<n){
           if(arr[i]>arr[j]){
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
           }
           j++;
       }
       i++;
   }
}
