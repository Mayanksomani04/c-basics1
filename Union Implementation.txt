Union Implementation 

#include <iostream>

using namespace std;

union student{
    int id;
    int roll_no;
    char name[50];
    int clas;
    
    
    void getdetails(){
        cout<<"Enter the id of student "<<endl;
        cin>>id;
        cout<<"Enter the rol_no of student "<<endl;
        cin>>roll_no;
        cout<<"Enter the name of student "<<endl;
        cin>>name;
        cout<<"Enter the class of student "<<endl;
        cin>>clas;
    }
    void printdetails(){
        cout<<"The id of student is "<<endl;
        cout<<id<<endl;
        cout<<"The roll_no of student is "<<endl;
        cout<<roll_no<<endl;
        cout<<"The name of student is "<<endl;
        cout<<name<<endl;
        cout<<"The class of student is "<<endl;
        cout<<clas<<endl;
    }
};
int main()
{
    int n;
    cout<<"Enter the no. of students ";
    cin>>n;
    student s;
    for(int i=0;i<n;i++){
        
        s.getdetails();
        s.printdetails();
    }

    return 0;
}
