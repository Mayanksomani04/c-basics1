Roman number convertor
#include <iostream>
using namespace std;

int main(){
    int n;
    int i;
    cout<<" Enter the value you want to convert ";
    cin>>n;
   string roman [] ={"M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I"};
   int value[]={1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1};
     string result = "";
   for(i=0;i<13;i++){
       while(n-value[i]>=0){
           result = result + roman[i];
           n=n-value[i];
       }
   }
   
   cout<<result;
    return 0;
}
