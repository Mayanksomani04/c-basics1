Prime no. In range
#include <iostream>

using namespace std;

int main()
{
    int start,end;
    cin>>start>>end;
   
    for(int i=start+1;i<end;i++){
         int count=0;
        for(int j=2;j<i;j++){
           if(i%j==0){
             count++; 
           }
        }
        if(count==0){
            cout<<i;
        }
    }

    return 0;
}
