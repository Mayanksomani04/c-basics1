Majority of element n/2
#include <iostream>
#include <cmath>

using namespace std;

int main(){
   int n;
   cout<<"Enter the size of array ";
   cin>>n;
   int arr[n];
   for(int i=0;i<n;i++){
       cin>>arr[i];
   }
   
   for(int i=0;i<n;i++){
       int count = 0;
       for(int j=0;j<n;j++){
         if(arr[i]==arr[j])
       {
          count++; 
          
       }
       }
       if(count>=n/2){
              cout<<"The majority element is "<<arr[i];
              break;
          }
   }
}
