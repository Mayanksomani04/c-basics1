Determine the root of the quadratic equation
#include <iostream>
#include <cmath>

using namespace std;

int main(){
   
int a,b,c,d;
cout<<"Enter the value of a,b,c and ";
cin>>a>>b>>c;
d=b*b-4*a*c;
if (d>=0) {
        float x1 = (-b + sqrt(d)) / (2*a);
       float x2 = (-b - sqrt(d)) / (2*a);
        cout<<"x1 = "<<x1<<endl;
        cout<<"x2 = "<<x2<<endl;
    }else{
        cout<<"No real value exist ";
    }
    return 0;
}
