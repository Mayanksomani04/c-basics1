Calculator using switch case

#include <iostream>

using namespace std;

int main(){
    
  float a,b;
    cout<<"Enter the value of a and b ";
    cin>>a>>b;
    char Op;
   cout<<"Enter the operator +,-,/,* ";
   cin>>Op;
   switch(Op){
       case '+':
       cout<<a<<"+"<<b<<"="<<a+b;
       break;
       case '-':
       cout<<a<<"-"<<b<<"="<<a-b;
       break;
       case '/':
       cout<<a<<"/"<<b<<"="<<a/b;
       break;
       case '*':
       cout<<a<<"*"<<b<<"="<<a*b;
       break;
       default:
       cout<<"operator is not correct ";
       break;
   }

    return 0;
}
