Binary Search
#include <iostream>

using namespace std;

int main()
{
   int n;
   cout<<"Enter the size of array ";
   cin>>n;
   int arr[n];
   cout<<"Enter the elements of array "; 
   for(int i=0;i<n;i++){
       cin>>arr[i];
   }
   int key;
   cout<<"Enter the key element to search ";
   cin>>key;
   int start=0,end=n-1,mid;
   mid=(start+end)/2;
   while(start<=end){
       if(key>arr[mid]){
           start=mid+1;
       }else if(key<arr[mid]){
           end=mid-1;
       }else{
           cout<<"The element at : "<<mid+1;
           break;
       }
       mid=(start+end)/2;
   }
   if(start>end){
       cout<<"The item is not in array ";
   }
    return 0;
}
